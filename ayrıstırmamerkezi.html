Sets
    i   "Firmalar / AM adaylari" /f1*f10/
    t   "Toplama Merkezleri"     /tm1, tm2/;

Parameters
    load(t)        "Toplama merkezlerinin toplam yukleri"
    distTM(i,t)    "TM -> firma mesafesi (km)";


load('tm1') = 51;

Binary Variables
    y(i);

Variable
    z;

Equations
    obj
    oneAM;


obj..
    z =e= sum((i,t), load(t) * distTM(i,t) * y(i));


oneAM..
    sum(i, y(i)) =e= 1;


Model AM_pmedian_TM /all/;
Solve AM_pmedian_TM using MIP minimizing z;

Display z.l, y.l;


Scalar z_check;
z_check = load('tm1')*distTM('f8','tm1') + load('tm2')*distTM('f8','tm2');
